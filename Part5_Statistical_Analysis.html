#  Q1. What is the average, median, and mode of transaction amounts?
#  Goal: Understand central tendency of how much people are transacting.
from scipy import stats
import pandas as pd
df = pd.read_csv("bank_transactions_data.csv")
mean_amount = df['TransactionAmount'].mean()
median_amount = df['TransactionAmount'].median()
mode_amount = df['TransactionAmount'].mode()[0]  # sabse pehla mode

print("Average Transaction Amount:", mean_amount)
print("Median Transaction Amount:", median_amount)
print("Mode Transaction Amount:", mode_amount)

Average Transaction Amount: 297.5937778662421
Median Transaction Amount: 211.14
Mode Transaction Amount: 11.46

# Q6. What is the skewness and kurtosis of the transaction amount?
from scipy.stats import skew, kurtosis

kurt = kurtosis(df['TransactionAmount'])

print("Skewness of Transaction Amount:", skewness)
print("Kurtosis of Transaction Amount:", kurt)
